# --- user-accounts (auth subpackage; server-side only) ---
# Password hashing for the credential store. argon2-cffi ships prebuilt wheels (no compiler
# needed on Windows) and is a current, well-vetted KDF. Used ONLY inside src/auth/ — never logs
# or returns plaintext (BACKEND_EXECUTION_CONTRACT §4).
argon2-cffi
# Google OAuth Authorization-Code flow (server-side, config-gated OFF when creds absent —
# BACKEND_EXECUTION_CONTRACT §6). Authlib's requests client handles the code→token exchange;
# the browser never sees the client secret or Google tokens.
authlib

# --- byo-ai-key (per-user encrypted AI key; server-side only) ---
# Symmetric, decryptable encryption-at-rest for the per-user Anthropic key (Fernet). The key must
# RECOVER to plaintext server-side to call Anthropic, so this is encryption (NOT hashing like
# argon2). Keyed by the gitignored env AI_KEY_ENCRYPTION_KEY (absent ⇒ ephemeral per-process key,
# no crash). Used ONLY inside src/auth/crypto.py — never logs/returns the key (byo-ai-key §4/§7).
cryptography

# --- persistent-db (Postgres auth store; server-side only) ---
# psycopg 3, the SYNC Postgres driver (the auth ports are sync -- NO async). [binary] ships a
# prebuilt wheel so no local C build toolchain is needed on Windows / a slim image. Used ONLY
# inside src/auth/postgres_store.py (the auth leaf) when ACCOUNT_STORE=postgres; in-memory stays
# the default. The DB only ever sees ciphertext + the masked last4 (persistent-db ARCHITECTURE 3.4/8).
psycopg[binary]
# The connection pool (psycopg_pool.ConnectionPool) -- one process-wide pool replaces the in-memory
# adapter module lock (ARCHITECTURE 1.3). No ORM, no migration framework (idempotent bootstrap DDL).
psycopg-pool

# --- ai-recommendations (in-app LLM proxy; server-side only) ---
# The Anthropic SDK for the in-app AI recommendation call (src/core/ai_recommendation.py ->
# AnthropicLLMProvider, claude-opus-4-8). It is LAZY-imported: when the package is ABSENT the proxy
# silently degrades to the deterministic StubLLMProvider (canned "[STUB]" answers, NOT live model
# output), so it MUST be installed for real recommendations. The key is read server-side only (the
# user's own key from the auth store, or the shared ANTHROPIC_API_KEY) and never reaches the browser.
anthropic
